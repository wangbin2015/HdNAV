var searchOption = new HDMap.GeocoderSearchOptions('http://190.0.0.29:8080/', function (result) {
    var infoList=result.getInfoList();
    var marker = new HDMap.marker([infoList[i].location.lat, infoList[i].location.lon]);
    map.addLayer(marker);
    //绑定popup
    marker.bindPopup("<p>"+infoList[i].addrname+"</p>");
    //打开popup
    marker.openPopup();
    }
});
//获取DOM节点
var text=document.getElementById('inputtext');

//绑定点击事件
btn.onclick=function(){
    //创建GeocoderSearch服务
    var geocoderSearcher = new HDMap.GeocoderSearch(searchOption);

}

创建（Creation）

geocoder创建 说明
var geo = new HDMap.Geocoder(); 实例化一个Geocoder对象

选项（Options）

选项 类型 说明
address Sring 搜索的关键字
region Sring 行政区划的编码，只能支持市以下的行政区划地理范围查询
location function 搜索的经纬度
url Sring 地图服务端口地址
onSearchComplete function 检索结束后的回调函数

方法（Methods）

选项 类型 说明
geoEncode() jsonObject 地理编码
geoDecode() jsonObject 逆地理编码
